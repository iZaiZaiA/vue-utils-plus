# 日期时间

各种日期时间操作的方法，更多 js 原生方法，查阅：Date在新窗口打开

import { utilsDate } from "vue-utils-plus"

# 取当前时间

getDateTime() 支持版本：V0.0.3

# 返回内容

参数名说明
dateTime日期时间
date日期
time时间
obj单独

# code示例

const { getDateTime } = utilsDate()

getDateTime()

# 当前时间戳

dateNow() 支持版本：V0.0.3

# 返回内容

参数字段说明
-当前时间戳

# code示例

const { dateNow } = utilsDate()

dateNow()

# 日期时间格式化

dateFormat() 支持版本：V0.0.2

# 参数说明

参数类型可选值默认值说明
date---日期对象
format--yyyy-MM-dd hh:mm:ss格式

# 返回内容

参数字段说明
-格式化后的内容

# code示例

const { dateFormat } = utilsDate()

dateFormat(new Date(), 'yyyy-MM-dd') // 2020-01-01

# 计算时间

calcDate() 支持版本：V0.0.2

# 参数说明

参数类型可选值默认值说明
date1---日期时间1
date2---日期时间2

# 返回内容

参数名说明
leave1计算天数后剩余的毫秒数
leave2计算小时数后剩余的毫秒数
leave3计算分钟数后剩余的毫秒数
days剩余天数
hours剩余小时
minutes剩余分钟
seconds剩余秒

# code示例

const { calcDate } = utilsDate()

calcDate('2022-07-29 10:22:14', '2022-08-14 23:59:59')
