# 文件处理

各种文件处理操作的方法

import { utilsFile } from "vue-utils-plus"

# 判断文件大小

isSize() 支持版本：V0.0.5

# 参数说明

参数类型可选值默认值说明
fileSize---文件数据的字节
size---要判断的大小, 单位M

# 返回内容

类型说明
Boolean返回 true 或 false

# code示例

const { isSize } = utilsFile()

isSize(102400,10) // true

# base64转成文件

base64ToFile() 支持版本：V0.0.6

# 参数说明

参数类型可选值默认值说明
base64---文件base64数据
type--image/jpeg文件类型，详情可以参阅 blob文件类型在新窗口打开
fileName--当前时间戳文件名称，默认为当前时间戳
suffix--"jpg"文件后缀

# 返回内容

参数名说明
blob新的文件对象

# code示例

const { base64ToFile } = utilsFile()

const fileBase64 = '文件base64数据'

const blob = base64ToFile(fileBase64)

//封装为formData进行请求
let formData = new FormData();
formData.append("file", blob);

console.log(formData)

# 下载文件

downloadBlob() 支持版本：V0.0.2

# 参数说明

参数类型可选值默认值说明
data---文件数据，通常为接口返回的 data
disposition---headers['content-disposition']
type--application/vnd.ms-exceltype是文件类，详情可以参阅 blob文件类型在新窗口打开

# code示例

const { downloadBlob } = utilsFile()

downloadBlob('接口相关数据')
